**大数减法**：
大数运算的减法与加法类似，实际上也是通过模拟列竖式来进行计算的，只是把‘+’号变成了‘-’号，进位变成了借位。

基本思想：

* 先判断数A和B的大小，用大的数减小的数
* 其余计算过程与加法类似，需要注意借位与进位的区别

虽然代码看起来复杂，但是思路简单

参考代码：
```c++
#include<string>
#include<cstdio>
#include<iostream>
#include<vector>
#include<stack>
#include<cstring>
using namespace std;

int main(){
  int t;
	cin>>t;
	string strA, strB;
	for(int i=1;i<=t;++i){
		cin>>strA>>strB;
		stack<int> stA, stB;
		for(int j=0;j<strA.size();++j) stA.push(strA[j]-48);
		for(int j=0;j<strB.size();++j) stB.push(strB[j]-48);
		int borrow=0;
		vector<int> ans;
		int topA, topB;
		if(strA.size()>strB.size()){
			while(!stA.empty() && !stB.empty()){
				topA = stA.top();
				topB = stB.top();
				stA.pop();
				stB.pop();
				if(topA+borrow>=topB){
					ans.push_back(topA+borrow-topB);
					borrow = 0;
				}else{
					ans.push_back(topA+10+borrow-topB);
					borrow = -1;
				}
			}	
			while(!stA.empty()){
				ans.push_back(stA.top()+borrow);
				stA.pop();
				borrow = 0;
			}	
			cout<<"Case "<<i<<":"<<endl;
			for(int j=0;j<strA.size();++j){
				cout<<strA[j];
			}
			cout<<" - ";
			for(int j=0;j<strB.size();++j) cout<<strB[j];
			cout<<" = ";
			int pos=ans.size()-1;
			while(ans[pos]==0) --pos;
      for(int j=pos;j>=0;--j) cout<<ans[j];
			cout<<endl<<endl;
		}else if(strA.size()<strB.size()){
			while(!stA.empty() && !stB.empty()){
				topA = stA.top();
				topB = stB.top();
				stA.pop();
				stB.pop();
				if(topB+borrow>=topA){
					ans.push_back(topB+borrow-topA);
					borrow = 0;
				}else{
					ans.push_back(topB+10+borrow-topA);
					borrow = -1;
				}
			}	
			while(!stB.empty()){
				ans.push_back(stB.top()+borrow);
				stB.pop();
				borrow = 0;
			}	
			cout<<"Case "<<i<<":"<<endl;
			for(int j=0;j<strA.size();++j) cout<<strA[j];
			cout<<" - ";
			for(int j=0;j<strB.size();++j) cout<<strB[j];
			cout<<" = -";
			int pos=ans.size()-1;
			while(ans[pos]==0) --pos;
      for(int j=pos;j>=0;--j) cout<<ans[j];
			cout<<endl<<endl;
		}else{
			if(strcmp(strA.c_str(), strB.c_str())>=0){
				while(!stA.empty() && !stB.empty()){
					topA = stA.top();
					topB = stB.top();
					stA.pop();
					stB.pop();
					if(topA+borrow>=topB){
						ans.push_back(topA+borrow-topB);
						borrow = 0;
					}else{
						ans.push_back(topA+10+borrow-topB);
						borrow = -1;
					}
				}		
				cout<<"Case "<<i<<":"<<endl;
				for(int j=0;j<strA.size();++j) cout<<strA[j];
				cout<<" - ";
				for(int j=0;j<strB.size();++j) cout<<strB[j];
				cout<<" = ";
				int pos=ans.size()-1;
				while(ans[pos]==0) --pos;
        for(int j=pos;j>=0;--j) cout<<ans[j];
				cout<<endl<<endl;
			}else{
				while(!stA.empty() && !stB.empty()){
					topA = stA.top();
					topB = stB.top();
					stA.pop();
					stB.pop();
					if(topB+borrow>=topA){
						ans.push_back(topB+borrow-topA);
						borrow = 0;
					}else{
						ans.push_back(topB+10+borrow-topA);
						borrow = -1;
					}
				}	
				cout<<"Case "<<i<<":"<<endl;
				for(int j=0;j<strA.size();++j) cout<<strA[j];
				cout<<" - ";
				for(int j=0;j<strB.size();++j) cout<<strB[j];
				cout<<" = -";
				int pos=ans.size()-1;
				while(ans[pos]==0) --pos;
        for(int j=pos;j>=0;--j) cout<<ans[j];
				cout<<endl<<endl;
			}
		}
	}
	return 0;
}
```
