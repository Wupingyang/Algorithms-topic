**大数幂运算**：

幂运算的结果通常很大，很容易超出int，long等基本类型的范围，因此不同使用基本类型表示出结果。在整数幂运算的基础上，进一步考虑浮点数的幂运算，例如$98.123^20$。

**基本思想**：

* 字符串表示浮点数，并去除小数点，当成整数先计算整数幂
  * 每一次相乘，都要调用大数乘法与大数加法
  * 循环相乘$n$次，$n$是幂次方
* 根据小数点位置和$n$值以及浮点数是否是否小于1，判断最终结果中小数点的位置

**参考代码**：

```c++
#include<string>
#include<cstdio>
#include<iostream>
#include<vector>
#include<stack>
#include<cstring>
#include<set>
#include<cmath>
using namespace std;

string bigNumMul(string strA, string strB){
	vector<int> add_res;
	for(int i=0;i<strB.size();++i){
		/*乘法计算 */ 
		int num = strB[i]-48;
		stack<int> st1;
		for(int j=0;j<strA.size();++j) st1.push(strA[j]-48);
		stack<int> st2;
		int mul_carry=0;
		while(!st1.empty()){
			int cur = st1.top();
			st1.pop();
			int mul_temp = cur * num + mul_carry;
			int digit = mul_temp % 10;
			mul_carry = mul_temp / 10;;
			st2.push(digit);
		} 
		if(mul_carry) st2.push(mul_carry);
	 
		vector<int> mul_res;
		while(!st2.empty()){
			mul_res.push_back(st2.top());
			st2.pop();
		}
		int offset = strB.size()-i-1;
		for(int j=0;j<offset;++j) mul_res.push_back(0);
		/*加法计算*/ 
		if(add_res.size()==0){
			for(auto mul:mul_res) add_res.push_back(mul);		
		}else{
			stack<int> st_add, st_mul, st_ans;
			for(auto add:add_res) st_add.push(add);
			for(auto mul:mul_res) st_mul.push(mul);
			int add_carry = 0;
			while(!st_mul.empty()){
				int cur_add = st_add.top();
				st_add.pop();
				int cur_mul = st_mul.top();
				st_mul.pop();
				int temp = cur_add + cur_mul + add_carry;
				int digit = temp % 10;
				add_carry = temp / 10;
				st_ans.push(digit);
			}
			while(!st_add.empty()){
				int cur_add = st_add.top();
				st_add.pop();
				int temp =  cur_add + add_carry;
				int digit = temp % 10;
				add_carry = temp / 10;
				st_ans.push(digit);
			}
			add_res.clear();
			while(!st_ans.empty()){
				add_res.push_back(st_ans.top());
				st_ans.pop();
			}
		}
	}
	string final="";
	for(auto x:add_res){
		final += to_string(x);
	}
	return final;	
}

int main(){
	string str;
	int n;
	while(cin>>str>>n){
		int dot_pos = str.find('.');
		str.erase(dot_pos, 1);//删除小数点，当成整数计算 
		string final = str;
		for(int i=1;i<n;++i){
			final=bigNumMul(str, final);
		}
		
		if(str[0]=='0'){//0.0432等小于1的小数表示 
			cout<<'.';	
			final.erase(0, 1);
		}else{
			int offset = final.size()-(str.size()-dot_pos)*n;
			final.insert(offset, ".");
		}
		while(final.back()=='0'){//消除末尾的0 
			final.erase(final.size()-1);
		}
		cout<<final<<endl;
	}
    return 0;
}
```
